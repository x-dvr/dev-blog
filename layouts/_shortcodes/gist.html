{{- /* Fetch a GitHub gist at build time and render via Hugo's chroma so it inherits the site's syntax-highlighting theme (and light/dark toggle). Usage: {{< gist user id [file] >}} */ -}} {{- $id := .Get 1 -}} {{- $only := .Get 2 -}} {{- $apiURL := printf "https://api.github.com/gists/%s" $id -}} {{- with resources.GetRemote $apiURL -}} {{- $data := . | transform.Unmarshal -}}
{{- range $name, $f := $data.files -}} {{- if or (not $only) (eq $name $only) -}} {{- $lang := lower (or $f.language "") -}}
{{ transform.Highlight $f.content $lang "" }}
{{ $name }} view on GitHub
{{- end -}} {{- end -}}
{{- else -}} {{ errorf "gist %q fetch failed" $id }} {{- end -}}
